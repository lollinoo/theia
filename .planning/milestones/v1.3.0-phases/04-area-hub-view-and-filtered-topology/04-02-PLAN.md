**1. Watermark.tsx** -- Simple atmospheric component per D-17/D-18/D-19:

```typescript
import type { Area } from '../types/api';

interface WatermarkProps {
  selectedAreaId: string | null;
  areas: Area[];
}

export function Watermark({ selectedAreaId, areas }: WatermarkProps) {
  const text = selectedAreaId
    ? (areas.find((a) => a.id === selectedAreaId)?.name ?? 'AREA').toUpperCase()
    : 'GLOBAL TOPOLOGY';

  return (
    <div
      className="fixed bottom-6 left-6 z-0 pointer-events-none select-none"
      aria-hidden="true"
    >
      <span
        className="font-sans font-semibold text-2xl tracking-tight
                   text-on-bg-muted opacity-[0.06] dark:opacity-[0.12]
                   transition-opacity duration-150"
      >
        {text}
      </span>
    </div>
  );
}
```

Per D-17: ~1.5rem / text-2xl per locked decision, Outfit font (font-sans in token system), very low opacity.
Per D-18: "GLOBAL TOPOLOGY" on Hub/Global, area name uppercase on filtered canvas.
Per D-19: 150ms fade transition via `transition-opacity duration-150`.
