  1. Create `frontend/src/types/metrics.ts`:
     ```typescript
     export interface DeviceMetricsDTO {
       device_id: string;
       cpu_percent: number | null;
       mem_percent: number | null;
       temp_celsius: number | null;
       uptime_secs: number | null;
       collected_at: string;
     }

     export interface LinkMetricsDTO {
       device_id: string;
       if_name: string;
       tx_bps: number | null;
       rx_bps: number | null;
       utilization: number | null;
       collected_at: string;
     }

     export interface AlertDTO {
       device_id: string;
       severity: string;  // "warning" | "critical"
       alert_name: string;
       state: string;     // "firing" | "resolved"
       summary: string;
     }

     export type AlertStatus = 'normal' | 'degraded' | 'down';

     export interface SnapshotPayload {
       device_metrics: Record<string, DeviceMetricsDTO>;
       link_metrics: Record<string, LinkMetricsDTO[]>;
       alerts: AlertDTO[];
     }

     export interface WSMessage {
       type: 'snapshot' | 'metrics' | 'link_metrics' | 'alert';
       payload: unknown;
     }
     ```

     Add helper functions:
     - `formatUptime(secs: number): string` -- "14d 3h" or "2h 15m" or "45m" short format
     - `metricColor(value: number): string` -- green (<60%), yellow (60-85%), red (>85%) as Tailwind class names (text-status-up, text-yellow-400, text-status-down)
     - `utilizationColor(value: number): string` -- returns hex color for link stroke: green=#00c853 (<0.5), yellow=#ffc107 (0.5-0.8), red=#ff1744 (>0.8)
     - `alertStatusForDevice(deviceId: string, alerts: AlertDTO[]): AlertStatus` -- returns 'down' if any critical alert firing, 'degraded' if any warning alert firing, 'normal' otherwise
     - `formatThroughput(bps: number): string` -- similar to formatBandwidth but for live values. Show "1.2 Mbps" with one decimal for values >= 1M, "456 Kbps" for >= 1K, "0 bps" for 0.

  2. Create `frontend/src/hooks/useWebSocket.ts`:
     - `useWebSocket(url: string)` returns `{ snapshot, connected, reconnecting }`
     - `snapshot` is `SnapshotPayload | null` -- the latest full state
     - Manages WebSocket lifecycle:
       a. Connect on mount
       b. Parse incoming JSON messages
       c. On "snapshot" message, update snapshot state
       d. On close/error, set reconnecting=true, attempt reconnect with exponential backoff (1s, 2s, 4s, 8s, max 30s)
       e. On successful reconnect, reset backoff, set reconnecting=false
       f. Cleanup on unmount
     - Derive WebSocket URL from window.location: `ws://${window.location.host}${url}`
     - Use useRef for the WebSocket instance to avoid stale closures
     - Use useCallback for message handler

  3. Create `frontend/src/components/ReconnectBanner.tsx`:
     - Simple component: `ReconnectBanner({ visible }: { visible: boolean })`
     - When visible: fixed top-center banner, semi-transparent, "Reconnecting..." text with a subtle spinner
     - Tailwind classes: `fixed top-4 left-1/2 -translate-x-1/2 z-50 rounded-lg bg-yellow-900/80 px-4 py-2 text-sm text-yellow-200 backdrop-blur-sm transition-opacity duration-300`
     - Fade in/out with opacity transition
