**Part A: Create useBridgeHealth hook (per D-13, D-14, D-15 — BRIDGE-05)**

Create `frontend/src/hooks/useBridgeHealth.ts`:

```typescript
import { useEffect, useState } from 'react';

const BRIDGE_HEALTH_URL = 'http://localhost:1337/health';
const POLL_INTERVAL_MS = 30_000;

export function useBridgeHealth(): { bridgeRunning: boolean } {
  const [bridgeRunning, setBridgeRunning] = useState(false);

  useEffect(() => {
    let cancelled = false;

    async function check() {
      try {
        const resp = await fetch(BRIDGE_HEALTH_URL);
        if (!cancelled) setBridgeRunning(resp.ok);
      } catch {
        if (!cancelled) setBridgeRunning(false);
      }
    }

    void check();
    const id = window.setInterval(() => { void check(); }, POLL_INTERVAL_MS);
    return () => {
      cancelled = true;
      window.clearInterval(id);
    };
  }, []);

  return { bridgeRunning };
}
```

Key requirements (per D-15):
- Silent failure: catch block calls `setBridgeRunning(false)` only — NO `console.error` or toast
- `cancelled` flag prevents state updates after unmount
- Polls on mount and every 30 seconds (per D-13)

Create `frontend/src/hooks/useBridgeHealth.test.ts`:

```typescript
import { describe, it, expect, vi, beforeEach, afterEach } from 'vitest';
import { renderHook, act, waitFor } from '@testing-library/react';
import { useBridgeHealth } from './useBridgeHealth';

describe('useBridgeHealth', () => {
  beforeEach(() => {
    vi.useFakeTimers();
    vi.stubGlobal('fetch', vi.fn());
  });

  afterEach(() => {
    vi.useRealTimers();
    vi.restoreAllMocks();
  });

  it('returns bridgeRunning=true when health check succeeds', async () => {
    (global.fetch as ReturnType<typeof vi.fn>).mockResolvedValue({ ok: true });
    const { result } = renderHook(() => useBridgeHealth());
    await waitFor(() => expect(result.current.bridgeRunning).toBe(true));
  });

  it('returns bridgeRunning=false when health check fails', async () => {
    (global.fetch as ReturnType<typeof vi.fn>).mockRejectedValue(new Error('network'));
    const { result } = renderHook(() => useBridgeHealth());
    await waitFor(() => expect(result.current.bridgeRunning).toBe(false));
  });

  it('returns bridgeRunning=false when response is not ok', async () => {
    (global.fetch as ReturnType<typeof vi.fn>).mockResolvedValue({ ok: false });
    const { result } = renderHook(() => useBridgeHealth());
    await waitFor(() => expect(result.current.bridgeRunning).toBe(false));
  });

  it('polls on 30s interval', async () => {
    (global.fetch as ReturnType<typeof vi.fn>).mockResolvedValue({ ok: true });
    renderHook(() => useBridgeHealth());
    await act(async () => { await vi.advanceTimersByTimeAsync(0); });
    expect(global.fetch).toHaveBeenCalledTimes(1);
    await act(async () => { await vi.advanceTimersByTimeAsync(30_000); });
    expect(global.fetch).toHaveBeenCalledTimes(2);
  });

  it('cleans up interval on unmount', async () => {
    (global.fetch as ReturnType<typeof vi.fn>).mockResolvedValue({ ok: true });
    const { unmount } = renderHook(() => useBridgeHealth());
    await act(async () => { await vi.advanceTimersByTimeAsync(0); });
    unmount();
    await act(async () => { await vi.advanceTimersByTimeAsync(60_000); });
    // After unmount + 60s, fetch should have been called only once (initial) or twice (initial + one interval before unmount)
    expect((global.fetch as ReturnType<typeof vi.fn>).mock.calls.length).toBeLessThanOrEqual(2);
  });
});
```

| Boundary | Description |
|----------|-------------|
| Frontend → Backend API | WinBox credentials fetched via authenticated-context API call |
| Frontend → Bridge (localhost:1337) | Cross-origin POST with credentials (ip, username, password) — bridge is on localhost |
| Bridge → WinBox binary | Bridge launches WinBox with credentials as CLI args — bridge validates origin |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-25-07 | I (Information Disclosure) | WinBox credential fetch | mitigate | fetchWinBoxCredentials response (ip, username, password) is held in local function scope only — never stored in React state, never logged. Credentials are immediately forwarded to bridge and then garbage collected. |
| T-25-08 | T (Tampering) | Bridge launch POST | accept | The POST to localhost:1337/launch sends credentials over unencrypted HTTP. This is accepted because: (a) traffic is localhost-only, (b) bridge validates Origin+Host headers (Phase 26, BRIDGE-03), (c) TLS on localhost requires cert complexity explicitly deferred in requirements. |
| T-25-09 | S (Spoofing) | Bridge health check | accept | Health check is a simple GET to localhost:1337/health with no auth. A malicious local process could spoof the bridge on port 1337. Accepted because: bridge is a locally-installed user binary, and Phase 26 adds Origin+Host dual validation (BRIDGE-03) for all non-health endpoints. |
| T-25-10 | D (Denial of Service) | useBridgeHealth polling | mitigate | 30-second interval (not aggressive). Silent catch prevents console spam. Cleanup on unmount prevents leaked intervals. |
| T-25-11 | I (Information Disclosure) | Console logging | mitigate | No console.error or console.log in useBridgeHealth or bridge launch catch blocks (per D-15). Network errors to localhost:1337 are expected and silenced. |
</threat_model>
