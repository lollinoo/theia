The new Makefile section should be:

```makefile
BRIDGE_OUT := bridge_binaries
BRIDGE_SRC := ./cmd/winbox-bridge/

# Windows and Linux: CGO_ENABLED=0 (fyne.io/systray is pure Go on these platforms)
# macOS: requires CGO_ENABLED=1 (Cocoa via Objective-C) — build natively on Mac or via CI
BRIDGE_TARGETS_NOCGO := windows/amd64 windows/arm64 linux/amd64 linux/arm64

bridge-build-all: ## Cross-compile winbox-bridge for Windows + Linux (macOS requires native Mac — use CI)
	@rm -rf $(BRIDGE_OUT)
	@mkdir -p $(BRIDGE_OUT)
	@for target in $(BRIDGE_TARGETS_NOCGO); do \
		os=$$(echo $$target | cut -d/ -f1); \
		arch=$$(echo $$target | cut -d/ -f2); \
		ext=""; \
		ldextra=""; \
		if [ "$$os" = "windows" ]; then ext=".exe"; ldextra="-H=windowsgui"; fi; \
		output="$(BRIDGE_OUT)/winbox-bridge-$${os}-$${arch}$${ext}"; \
		echo "Building $$output ..."; \
		CGO_ENABLED=0 GOOS=$$os GOARCH=$$arch go build -ldflags="-s -w $${ldextra}" -o "$$output" $(BRIDGE_SRC) || exit 1; \
	done
	@echo ""
	@echo "Bridge binaries built in $(BRIDGE_OUT)/:"
	@ls -la $(BRIDGE_OUT)/
	@echo ""
	@echo "NOTE: macOS binaries (darwin/amd64, darwin/arm64) require CGO_ENABLED=1."
	@echo "      Build natively on Mac: CGO_ENABLED=1 GOOS=darwin GOARCH=arm64 go build -ldflags=\"-s -w\" -o $(BRIDGE_OUT)/winbox-bridge-darwin-arm64 $(BRIDGE_SRC)"
```

- `ldextra` variable adds `-H=windowsgui` for Windows only
- Help text updated to clarify macOS limitation
- Echo note at end explains how to build macOS locally

Replace the `build-bridge` job (lines 222-268) with a version that splits into per-platform matrix entries with different runners and CGO settings.

The new job structure uses a matrix with `include` entries that specify `runs-on`, `cgo`, and `ldflags` per target:

```yaml
  build-bridge:
    name: "Bridge: ${{ matrix.goos }}/${{ matrix.goarch }}"
    if: startsWith(github.ref, 'refs/tags/v')
    strategy:
      matrix:
        include:
          - goos: windows
            goarch: amd64
            ext: .exe
            runs-on: ubuntu-latest
            cgo: "0"
            ldflags: "-s -w -H=windowsgui"
          - goos: windows
            goarch: arm64
            ext: .exe
            runs-on: ubuntu-latest
            cgo: "0"
            ldflags: "-s -w -H=windowsgui"
          - goos: linux
            goarch: amd64
            ext: ""
            runs-on: ubuntu-latest
            cgo: "0"
            ldflags: "-s -w"
          - goos: linux
            goarch: arm64
            ext: ""
            runs-on: ubuntu-latest
            cgo: "0"
            ldflags: "-s -w"
          - goos: darwin
            goarch: amd64
            ext: ""
            runs-on: macos-latest
            cgo: "1"
            ldflags: "-s -w"
          - goos: darwin
            goarch: arm64
            ext: ""
            runs-on: macos-latest
            cgo: "1"
            ldflags: "-s -w"
    runs-on: ${{ matrix.runs-on }}
    steps:
      - uses: actions/checkout@v4

      - uses: actions/setup-go@v5
        with:
          go-version-file: go.mod

      - name: Build bridge binary
        env:
          CGO_ENABLED: ${{ matrix.cgo }}
          GOOS: ${{ matrix.goos }}
          GOARCH: ${{ matrix.goarch }}
        run: |
          mkdir -p bridge_binaries
          go build -ldflags="${{ matrix.ldflags }}" -o "bridge_binaries/winbox-bridge-${{ matrix.goos }}-${{ matrix.goarch }}${{ matrix.ext }}" ./cmd/winbox-bridge/

      - name: Upload release asset
        uses: softprops/action-gh-release@v2
        with:
          files: bridge_binaries/winbox-bridge-${{ matrix.goos }}-${{ matrix.goarch }}${{ matrix.ext }}
        env:
          GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}
```

Key changes from current:
1. `runs-on: ubuntu-latest` moved from job level to `runs-on: ${{ matrix.runs-on }}` (dynamic per matrix entry)
2. Matrix entries gain `runs-on`, `cgo`, and `ldflags` fields
3. macOS entries use `runs-on: macos-latest` and `cgo: "1"`
4. Windows entries add `ldflags: "-s -w -H=windowsgui"` (suppress console window)
5. Linux entries use `ldflags: "-s -w"` (unchanged)
6. `CGO_ENABLED` is now `${{ matrix.cgo }}` instead of hardcoded `0`
7. `-ldflags` uses `${{ matrix.ldflags }}` instead of hardcoded `"-s -w"`
8. Job name includes OS/arch for clearer CI display

Note on macOS cross-arch: `macos-latest` is ARM (M-series) since mid-2024. Building `darwin/amd64` from ARM mac works because Apple clang supports cross-arch compilation. Both `darwin/amd64` and `darwin/arm64` can be built on the same `macos-latest` runner.
