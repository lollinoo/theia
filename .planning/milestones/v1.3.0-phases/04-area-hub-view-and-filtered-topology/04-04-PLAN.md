**1. useAreaFilteredTopology.ts** -- A custom hook using `useMemo` for filtering:

```typescript
import { useMemo } from 'react';
import type { Device, Link } from '../../types/api';

interface FilteredTopology {
  filteredDevices: Device[];
  filteredLinks: Link[];
  ghostDevices: Device[];
}

export function useAreaFilteredTopology(
  devices: Device[],
  links: Link[],
  selectedAreaId: string | null,
): FilteredTopology {
  return useMemo(() => {
    // No filter = show everything (Global view)
    if (!selectedAreaId) {
      return { filteredDevices: devices, filteredLinks: links, ghostDevices: [] };
    }

    // Devices in the selected area (per D-14: unassigned devices excluded)
    const areaDeviceIds = new Set(
      devices
        .filter((d) => d.area_id === selectedAreaId)
        .map((d) => d.id),
    );
    const filteredDevices = devices.filter((d) => areaDeviceIds.has(d.id));

    // Links where at least one endpoint is in the area
    const filteredLinks = links.filter(
      (l) => areaDeviceIds.has(l.source_device_id) || areaDeviceIds.has(l.target_device_id),
    );

    // Ghost devices: remote endpoints of cross-area links
    const ghostDeviceIds = new Set<string>();
    for (const link of filteredLinks) {
      if (!areaDeviceIds.has(link.source_device_id)) {
        ghostDeviceIds.add(link.source_device_id);
      }
      if (!areaDeviceIds.has(link.target_device_id)) {
        ghostDeviceIds.add(link.target_device_id);
      }
    }
    const ghostDevices = devices.filter((d) => ghostDeviceIds.has(d.id));

    return { filteredDevices, filteredLinks, ghostDevices };
  }, [devices, links, selectedAreaId]);
}
```

**1. Wire useAreaFilteredTopology in Canvas.tsx:**

Import the hook:

After `useCanvasData` returns `devices`, `topologyLinks`, call the filter:

```typescript
const { filteredDevices, filteredLinks, ghostDevices } = useAreaFilteredTopology(
```

The existing `loadTopology` in `useCanvasData` builds nodes/edges from ALL devices/links. Now we need to build nodes/edges from the FILTERED set instead.

There are two approaches:
a) Modify `useCanvasData` to accept `selectedAreaId` and filter internally
b) Apply filtering AFTER useCanvasData, rebuilding nodes/edges from filtered data

Approach (b) is cleaner -- keep the data hook pure and apply filtering as a layer:

After `useCanvasData` provides nodes/edges, apply a `useMemo` that:
1. Filters nodes to only include filteredDevices + ghostDevices
2. Creates ghost nodes for ghostDevices (with `isGhost: true` in data)
3. Filters edges to only include filteredLinks
4. Rebuilds edges with correct handle sides for the filtered node set

Implementation:
```typescript
const { displayNodes, displayEdges } = useMemo(() => {
  if (!selectedAreaId) {
    // Global view: use all nodes and edges as-is
    return { displayNodes: nodes, displayEdges: edges };
  }

  const filteredDeviceIds = new Set(filteredDevices.map(d => d.id));
  const ghostDeviceIds = new Set(ghostDevices.map(d => d.id));

  // Keep existing nodes that are in filtered set
  const areaNodes = nodes.filter(n => filteredDeviceIds.has(n.id));

  // Create ghost nodes for cross-area link endpoints
  const ghostNodes: Node<DeviceNodeData>[] = ghostDevices.map(device => {
    const existingNode = nodes.find(n => n.id === device.id);
    // Position ghost node near its connected real node
    const connectedLink = filteredLinks.find(
      l => l.source_device_id === device.id || l.target_device_id === device.id
    );
    const connectedRealDeviceId = connectedLink
      ? (connectedLink.source_device_id === device.id
          ? connectedLink.target_device_id
          : connectedLink.source_device_id)
      : null;
    const connectedRealNode = connectedRealDeviceId
      ? areaNodes.find(n => n.id === connectedRealDeviceId)
      : null;

    // Place ghost 200px offset from connected real node (per RESEARCH recommendation)
    const basePos = existingNode?.position
      ?? connectedRealNode
        ? { x: connectedRealNode!.position.x + 200, y: connectedRealNode!.position.y }
        : { x: 0, y: 0 };

    return {
      id: device.id,
      type: 'device',
      position: existingNode?.position ?? basePos,
      data: {
        device,
        pinned: false,
        isGhost: true,
        onGhostClick: (deviceId: string) => {
          const clickedDevice = devices.find(d => d.id === deviceId);
          if (clickedDevice?.area_id) {
            // Navigate to the ghost device's area
            // This needs to bubble up to App.tsx via a callback
          }
        },
      },
    };
  });

  const allDisplayNodes = [...areaNodes, ...ghostNodes];

  // Filter edges to only those in filteredLinks
  const filteredLinkIds = new Set(filteredLinks.map(l => l.id));
  const areaEdges = edges.filter(e => filteredLinkIds.has(e.id));

  return { displayNodes: allDisplayNodes, displayEdges: areaEdges };
}, [selectedAreaId, nodes, edges, filteredDevices, filteredLinks, ghostDevices, devices]);
```
