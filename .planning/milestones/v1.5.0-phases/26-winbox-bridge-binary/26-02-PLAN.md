    2. Add a new section after the "Release workflow" section (after line 143, the last line of the file). Insert:

    ```makefile
    # ---------------------------------------------------------------------------
    # WinBox Bridge cross-compilation
    # ---------------------------------------------------------------------------
    BRIDGE_OUT := bridge_binaries
    BRIDGE_SRC := ./cmd/winbox-bridge/
    BRIDGE_TARGETS := windows/amd64 windows/arm64 linux/amd64 linux/arm64 darwin/amd64 darwin/arm64

    bridge-build-all: ## Cross-compile winbox-bridge for all 6 targets
    	@rm -rf $(BRIDGE_OUT)
    	@mkdir -p $(BRIDGE_OUT)
    	@for target in $(BRIDGE_TARGETS); do \
    		os=$$(echo $$target | cut -d/ -f1); \
    		arch=$$(echo $$target | cut -d/ -f2); \
    		ext=""; \
    		if [ "$$os" = "windows" ]; then ext=".exe"; fi; \
    		output="$(BRIDGE_OUT)/winbox-bridge-$${os}-$${arch}$${ext}"; \
    		echo "Building $$output ..."; \
    		CGO_ENABLED=0 GOOS=$$os GOARCH=$$arch go build -ldflags="-s -w" -o "$$output" $(BRIDGE_SRC) || exit 1; \
    	done
    	@echo ""
    	@echo "Bridge binaries built in $(BRIDGE_OUT)/:"
    	@ls -la $(BRIDGE_OUT)/
    ```

    Key details:
    - `CGO_ENABLED=0` on every build command (D-15)
    - `-ldflags="-s -w"` strips debug info for smaller binaries
    - Output directory is `bridge_binaries/` at repo root (D-13)
    - Binary names: `winbox-bridge-{os}-{arch}[.exe]` — matches `bridge_handler.go` naming convention exactly
    - The `|| exit 1` ensures the make target fails if any single compilation fails

    3. Add `bridge_binaries/` to `.gitignore` if it exists, or note that it should not be committed. Check if `.gitignore` exists first — if so, append `bridge_binaries/` to it.

    **CI changes (per D-14):**

    Add a new job `build-bridge` to `.github/workflows/ci.yml`. This job:
    - Runs only on tag push (same condition as the existing `release` job: `if: startsWith(github.ref, 'refs/tags/v')`)
    - Does NOT depend on `backend` or `frontend` test jobs (bridge is independent, has its own tests)

    Insert the new job AFTER the `release` job (after the last line of ci.yml). The job structure:

    ```yaml
      build-bridge:
        name: Bridge Binaries
        if: startsWith(github.ref, 'refs/tags/v')
        runs-on: ubuntu-latest
        strategy:
          matrix:
            include:
              - goos: windows
                goarch: amd64
                ext: .exe
              - goos: windows
                goarch: arm64
                ext: .exe
              - goos: linux
                goarch: amd64
                ext: ""
              - goos: linux
                goarch: arm64
                ext: ""
              - goos: darwin
                goarch: amd64
                ext: ""
              - goos: darwin
                goarch: arm64
                ext: ""
        steps:
          - uses: actions/checkout@v4

          - uses: actions/setup-go@v5
            with:
              go-version-file: go.mod

          - name: Build bridge binary
            env:
              CGO_ENABLED: 0
              GOOS: ${{ matrix.goos }}
              GOARCH: ${{ matrix.goarch }}
            run: |
              mkdir -p bridge_binaries
              go build -ldflags="-s -w" -o "bridge_binaries/winbox-bridge-${{ matrix.goos }}-${{ matrix.goarch }}${{ matrix.ext }}" ./cmd/winbox-bridge/

          - name: Upload release asset
            uses: softprops/action-gh-release@v2
            with:
              files: bridge_binaries/winbox-bridge-${{ matrix.goos }}-${{ matrix.goarch }}${{ matrix.ext }}
            env:
              GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}
    ```

    Key details:
    - Uses `softprops/action-gh-release@v2` (latest stable) to upload assets to the GitHub Release created by the tag
    - Each matrix entry builds one binary and uploads it — 6 parallel jobs
    - `CGO_ENABLED: 0` as environment variable (D-15)
    - `-ldflags="-s -w"` strips debug symbols for smaller download
    - Uses `secrets.GITHUB_TOKEN` (automatically available in GitHub Actions, no extra secret needed)
    - The job is independent of the `release` Docker image job — they run in parallel
